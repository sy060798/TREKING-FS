let dataList=[],currentEditId=null,chart; // INIT document.getElementById("upload").addEventListener("change", importExcel); document.getElementById("checkAll").addEventListener("change", e=>{ document.querySelectorAll("tbody input").forEach(c=>c.checked=e.target.checked); }); // TAB function showTab(tab){ document.querySelectorAll(".tab").forEach(t=>t.classList.remove("active")); document.getElementById(tab).classList.add("active"); if(tab==="pivot") generatePivot(); } // IMPORT function importExcel(e){ let file=e.target.files[0]; let reader=new FileReader(); reader.onload=evt=>{ let wb=XLSX.read(new Uint8Array(evt.target.result),{type:'array'}); dataList=[]; wb.SheetNames.forEach(s=>{ let json=XLSX.utils.sheet_to_json(wb.Sheets[s]); json.forEach(r=>{ let stb=parseInt(r.STB)||0; let dpp=200000+stb*50000; dataList.push({ id:r.ID||Math.random(), wo:r.WO||"", area:r.AREA||"", wotype:r["WO TYPE"]||"", stb:stb, dpp:dpp, amount:Math.round(dpp*1.11), remark:r.REMARK||"NOT PAID" }); }); }); renderTable(); }; reader.readAsArrayBuffer(file); } // RENDER (TIDAK DIUBAH) function renderTable(){ let tbody=document.querySelector("tbody"); tbody.innerHTML=""; dataList.forEach((d,i)=>{ tbody.innerHTML+=` ${i+1} ${d.id} ${d.wo} ${d.area} ${d.wotype} ${d.stb} ${d.dpp} ${d.amount} ${d.remark} `; }); } // EDIT (TETAP) function editData(id){ let d=dataList.find(x=>x.id==id); currentEditId=id; edit_wo.value=d.wo; edit_area.value=d.area; edit_stb.value=d.stb; edit_remark.value=d.remark; modalEdit.style.display="flex"; } // ✅ FIX EDIT MASSAL function editMassal(){ let checked=[...document.querySelectorAll("tbody input:checked")]; if(checked.length===0){ alert("gunakan checkbox dulu"); return; } // ambil id tetap sistem lama currentEditId=checked.map(c=>c.dataset.id); modalEdit.style.display="flex"; } // ✅ FIX SAVE (DUKUNG MASSAL) function saveEdit(){ // massal if(Array.isArray(currentEditId)){ dataList.forEach(d=>{ if(currentEditId.includes(String(d.id))){ d.remark=edit_remark.value||d.remark; } }); }else{ let d=dataList.find(x=>x.id==currentEditId); d.wo=edit_wo.value; d.area=edit_area.value; d.stb=parseInt(edit_stb.value)||0; d.dpp=200000+d.stb*50000; d.amount=Math.round(d.dpp*1.11); d.remark=edit_remark.value; } renderTable(); closeModal(); } // DELETE (TETAP) function hapusTerpilih(){ let ids=[...document.querySelectorAll("tbody input:checked")].map(c=>c.dataset.id); dataList=dataList.filter(d=>!ids.includes(String(d.id))); renderTable(); } // EXPORT (TETAP) function exportExcel(){ let ws=XLSX.utils.json_to_sheet(dataList); let wb=XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb,ws,"DATA"); XLSX.writeFile(wb,"data.xlsx"); } // PIVOT (TETAP) function generatePivot(){ let g={}; dataList.forEach(d=>{ g[d.area]=(g[d.area]||0)+d.amount; }); if(chart) chart.destroy(); chart=new Chart(chartPivot,{ type:'bar', data:{labels:Object.keys(g),datasets:[{data:Object.values(g)}]} }); } function cekUpdate(){alert("OK");} function closeModal(){modalEdit.style.display="none";}
